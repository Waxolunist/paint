  const generate_htmlcolors_hsl = function() {
    let colors = [];  
    colors.push('hsl(0, 0%, 0%)');
    colors.push('hsl(0, 0%, 50%)');
    colors.push('hsl(0, 0%, 100%)');    
    let h = 340;
    while(h >= 0) {
        colors.push('hsl(' + h + ', 100%, 50%)');
        h -= 20;
    }
    return colors;
  }
  
  export const htmlcolors_hsl = generate_htmlcolors_hsl();
